Class Caneta{
	
					ATRIBUTOS

	private $cor;

					CONSTRUTORES
/*

* public function __primeiroConstrutor(){}

* public function Caneta(){} 
*/	

	public function Caneta($mod, $cor){ 	//metodo construtor
		$this->setModelo($mod);		//passando pelo metodo
		$this->cor=$cor;		//ou passando diretamente
		$this->tampar();
	}

					METODOS
	public function getModelo(){
		return $this->modelo;
	}
	public function setModelo(mod){
		$this->modelo = mod;
	}
	
	public function getCor(){
		return $this->cor;
	}
	public function setCor(cor){
		$this->cor = cor;
	}
	
	public function tampar(){
		$this->tampada = true;
	}
}

ENCAPSULAMENTO

Ideia de capsula
Ex.: Uma pilha, que possuem várias particuladidades por dentro, tal como carvão, zinco, etc, onde ficam escondidas, encapsuladas, e apenas os polos positivos e negativos ficam expostos.

Conceito:
Ocultar partes independentes da implementação, pemitindo construir partes invisíveis ao mundo exterior.
